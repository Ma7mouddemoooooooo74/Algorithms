#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

void bubbleSort(int a[] , int n){
 bool flag = true;
 for(int i=0;i<n-1;i++){
   for(int j=0;j<n-i-1;j++){
    if(a[j]>a[j+1]){
        swap(a[j+1],a[j]);
        flag=false;
    }
   }
   if(flag==true){
        break;
    }
 }
}

void printArray(int ar[] , int n){
 for(int i=0;i<n;i++){
   cout<<ar[i]<<" ";
 }
}

int main()
 {
 int a[5]={5,2,1,3,4};
 bubbleSort(a,5);
 printArray(a,5);
return 0;
  }
